# AI狼人杀项目环境配置示例
# 将此文件重命名为 .env 并填入你的API密钥

# OpenAI API配置
OPENAI_API_KEY=your-openai-api-key-here
OPENAI_API_BASE=https://api.openai.com/v1  # 可选，使用自定义API端点

# 阿里云通义千问API配置
DASHSCOPE_API_KEY=your-dashscope-api-key-here

# 选择使用的LLM提供商 (openai / dashscope)
LLM_PROVIDER=openai

# 模型选择
OPENAI_MODEL=gpt-4  # 或 gpt-3.5-turbo
DASHSCOPE_MODEL=qwen-turbo  # 或 qwen-plus, qwen-max

# 游戏配置（可选）
GAME_LANGUAGE=zh  # zh: 中文, en: 英文
DEBUG_MODE=false
