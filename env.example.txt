# - ModelScope Token: https://www.modelscope.cn/my/myaccesstoken

# ============================================================
# 方案A：LLM用ModelScope + TTS用DashScope（推荐🌟免费）
# ============================================================
#
# 优势：
# - LLM 使用 ModelScope 免费推理服务
# - TTS 使用 DashScope Qwen3-TTS-Flash 高质量语音
# - 两个服务互不影响，配置灵活
#
# 配置步骤：
# 1. 获取 ModelScope Token (ms- 开头)
# 2. 获取 DashScope API Key (sk- 开头)
# 3. 分别填入下面的配置项

# LLM 推理服务（ModelScope - 免费）
LLM_PROVIDER=modelscope
OPENAI_API_KEY=ms-your-modelscope-token-here
OPENAI_API_BASE=https://api-inference.modelscope.cn/v1/
OPENAI_MODEL=Qwen/Qwen2.5-32B-Instruct

# TTS 语音服务（DashScope - 有免费额度）
DASHSCOPE_API_KEY=sk-your-dashscope-api-key-here

# ============================================================
# 方案B：全部使用DashScope（简化配置）
# ============================================================
#
# 如果你只想用一个 API Key，可以只配置 DashScope：
# 
# LLM_PROVIDER=dashscope
# OPENAI_API_KEY=sk-your-dashscope-api-key-here
# OPENAI_API_BASE=https://dashscope.aliyuncs.com/compatible-mode/v1
# OPENAI_MODEL=qwen-plus
# DASHSCOPE_API_KEY=sk-your-dashscope-api-key-here
# 
# 注意：这种方式 LLM 和 TTS 都会计费

# ANTHROPIC_API_KEY=sk-ant-your-api-key-here

# ============================================================
# LLM 参数配置
# ============================================================

LLM_TEMPERATURE=0.8     # 温度参数 (0.0-2.0)，越高越随机，越低越确定
LLM_MAX_TOKENS=500      # 单次回复最大token数

# ============================================================
# TTS 语音合成配置
# ============================================================

# TTS 提供商: dashscope, azure, elevenlabs
TTS_PROVIDER=dashscope

# 是否启用语音合成
TTS_ENABLED=true

# DashScope TTS 配置
# 音色选择（Qwen3-TTS-Flash 模型）
TTS_VOICE=Cherry
#
# 可选音色（女声）：
# - Cherry  : 甜美清晰 ⭐ 默认推荐
# - Bella   : 温柔优雅
# - Amy     : 亲切自然
# - Emma    : 活泼可爱
# - Cora    : 成熟稳重
# - Eva     : 知性优雅
# - Kate    : 清新甜美
# - Luna    : 温柔轻柔
# - Sara    : 干练明快
#
# 可选音色（男声）：
# - William : 磁性低沉
# - James   : 沉稳可靠
# - Thomas  : 年轻活力
# - Daniel  : 温和儒雅
#
# 注意：如果使用旧模型 sambert-zhichu-v1（不推荐），
# 音色应改为：zhixiaobai, zhixiaoxia, zhiyan, zhitian, zhigang

# TTS 模型（DashScope）
TTS_MODEL=qwen3-tts-flash

# 语速和音高（可选）
TTS_SPEED=1.0           # 语速 (0.5-2.0)，1.0为正常速度
TTS_PITCH=1.0           # 音高 (0.5-2.0)，1.0为正常音高

# ============================================================
# 游戏配置
# ============================================================

GAME_LANGUAGE=zh        # 游戏语言: zh=中文, en=英文
DEBUG_MODE=false        # 调试模式: true=详细日志, false=正常日志

# 游戏规则（可选，默认使用 config/default.json）
GAME_NUM_PLAYERS=9      # 玩家数量

# ============================================================
# Web 服务配置
# ============================================================

WEB_HOST=0.0.0.0        # 监听地址，0.0.0.0表示监听所有网卡
WEB_PORT=8000           # 后端服务端口

# CORS 配置（可选）
# CORS_ORIGINS=http://localhost:5173,http://localhost:3000

# ============================================================
# 数据库配置（可选）
# ============================================================

# 缓存配置
CACHE_ENABLED=true      # 是否启用缓存
CACHE_TTL=3600          # 缓存过期时间（秒）

# 向量数据库（实验性功能）
# VECTOR_DB_ENABLED=false

# ============================================================
# 视频生成配置（实验性功能）
# ============================================================

VIDEO_ENABLED=false     # 是否启用视频生成
# VIDEO_FPS=30
# VIDEO_WIDTH=1920
# VIDEO_HEIGHT=1080

# ============================================================
# 日志配置
# ============================================================

LOG_LEVEL=INFO          # 日志级别: DEBUG, INFO, WARNING, ERROR
# LOG_FILE=data/logs/werewolf.log
